00094 HDC CreateSW (const char* source)
00095     {
00096     struct stat st = {};
00097     if (stat (source, &st) != 0) return NULL;
00098     size_t size = (size_t) st.st_size;
00099 
00100     char* text = new (std::nothrow) char [size + 1] {};
00101     if (!text) return NULL;
00102 
00103     FILE* f = NULL;
00104     fopen_s (&f, source, "r"); if (!f) { delete[] text; return NULL; }
00105     fread (text, size, 1, f);
00106     fclose (f);

00109     for (char* s = strchr (text, '\n'); s; s = strchr (s+1, '\n')) lines++;

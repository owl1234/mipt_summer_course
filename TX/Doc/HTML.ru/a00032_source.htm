00045 void DrawMan   (int x, int y, int sizeX, int sizeY, COLORREF color,
00046                 double hand, double legs, double head, double twist)
00047     {

00050 
00051     txLine (x, y - (0.35 + twist) * sizeY, x, y - 0.7*sizeY);
00052 
00053     txLine (x, y - (0.35 + twist) * sizeY, x - (0.5 + legs) * sizeX, y);
00054     txLine (x, y - (0.35 + twist) * sizeY, x + (0.5 + legs) * sizeX, y);
00055 
00056     txLine (x, y - 0.65*sizeY, x - sizeX/2,   y - 0.4*sizeY);
00057     txLine (x, y - 0.65*sizeY, x + sizeX/1.2, y - (0.7 + hand) * sizeY);
00058 
00059     txCircle (x, y - sizeY + (0.3 + head) * sizeX, 0.3*sizeX);
